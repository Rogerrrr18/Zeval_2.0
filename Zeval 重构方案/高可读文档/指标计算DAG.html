能力维度 Schema

5 个 MVP 维度一次 LLM 调用，LLM 只输出类别标签，后端做固定映射。

▸ LLM 输出 JSON Schema 
{
  "goal_completion": {
    "verdict": "achieved|partial|failed",
    "reason": "string",
    "evidence_turns": [int, ...]
  },
  "instruction_following": {
    "level": "L1|L2|L3|L4|L5",
    "reason": "string",
    "evidence_turns": [int, ...]
  },
  "multi_turn_coherence": {
    "level": "L1|L2|L3|L4|L5",
    "reason": "string",
    "evidence_turns": [int, ...]
  },
  "response_quality": {
    "level": "L1|L2|L3|L4|L5",
    "reason": "string",
    "evidence_turns": [int, ...]
  },
  "safety_and_tone": {
    "safety_ok": true|false,
    "level": "L1|L2|L3|L4|L5",
    "reason": "string"
  }
}
